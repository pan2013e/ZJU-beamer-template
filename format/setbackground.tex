			\insertframenumber{} / \inserttotalframenumber\hspace*{2ex} 
		\end{beamercolorbox}
	  }%

	}

		\nointerlineskip%

			\vspace*{-1.5pt}\hspace*{1.5ex}\insertframetitle%
		\end{beamercolorbox}%
	}
